# 存档处理

脚本模组的存档管理指南。

## 模组数据存储

每个模组可以在游戏存档中存储自定义数据。

### 基础操作

```gdscript
# 保存数据
var my_data = {
    "mod_version": "1.0.0",
    "player_flags": {
        "special_event_seen": true,
        "bonus_applied": false
    },
    "custom_stats": {
        "total_kills": 150,
        "explored_planets": 24
    },
    "timestamp": Time.get_unix_time_from_system()
}
bridge.save_mod_data("my_mod", my_data)

# 读取数据
var loaded = bridge.load_mod_data("my_mod")
if loaded:
    bridge.log_info("模组数据已加载")
    bridge.log_info("版本: " + loaded.mod_version)

# 删除数据
bridge.delete_mod_data("my_mod")
```

## 存档钩子

```gdscript
func _ready():
    bridge.register_hook("on_save", self, "_on_save")
    bridge.register_hook("on_load", self, "_on_load")

func _on_save(data):
    bridge.log_info("游戏正在保存...")
    # 在保存前执行清理或状态同步
    bridge.save_mod_data("my_mod_runtime", {
        "active_effects": _get_active_effects(),
        "runtime_flags": _runtime_flags
    })

func _on_load(data):
    bridge.log_info("游戏正在加载...")
    var save_data = bridge.load_mod_data("my_mod_runtime")
    if save_data:
        _restore_runtime_state(save_data)
        bridge.log_info("运行时状态已恢复")

func _get_active_effects():
    # 收集当前生效的状态
    return []

func _restore_runtime_state(data):
    # 恢复运行时状态
    _runtime_flags = data.get("runtime_flags", {})
```

## 数据迁移

当模组版本更新时，可能需要迁移旧存档格式：

```gdscript
func _ready():
    bridge.register_hook("on_load", self, "_on_load")

func _on_load(data):
    var mod_data = bridge.load_mod_data("my_mod")

    if not mod_data:
        return

    var saved_version = mod_data.get("mod_version", "0.0.0")
    var current_version = "1.1.0"

    if saved_version != current_version:
        bridge.log_info("正在进行存档迁移: " + saved_version + " -> " + current_version)
        mod_data = _migrate_data(mod_data, saved_version, current_version)
        mod_data.mod_version = current_version
        bridge.save_mod_data("my_mod", mod_data)
        bridge.log_info("存档迁移完成")

func _migrate_data(data, from_version, to_version):
    if from_version == "1.0.0":
        # 1.0.0 -> 1.1.0 迁移
        data.new_field = data.old_field
        data.erase("old_field")
        bridge.log_info("迁移: 1.0.0 -> 1.1.0")
    return data
```

## 数据验证

```gdscript
func _validate_mod_data(data) -> bool:
    var required_fields = ["mod_version", "player_flags", "custom_stats"]

    for field in required_fields:
        if not data.has(field):
            bridge.log_warn("存档数据缺少字段: " + field)
            return false

    return true

func _on_load(data):
    var mod_data = bridge.load_mod_data("my_mod")
    if mod_data and _validate_mod_data(mod_data):
        bridge.log_info("数据验证通过")
    else:
        bridge.log_warn("数据损坏，使用默认值")
        bridge.save_mod_data("my_mod", _get_default_data())
```

## 多玩家数据

```gdscript
func save_player_data(player_id):
    var player_specific = {
        "player_id": player_id,
        "score": bridge.get_player_stat(player_id, "total_score"),
        "achievements": _get_player_achievements(player_id)
    }
    bridge.save_mod_data("player_" + str(player_id), player_specific)

func load_player_data(player_id):
    var key = "player_" + str(player_id)
    return bridge.load_mod_data(key)
```

## 注意事项

- 每个模组的数据大小应控制在合理范围内（建议 < 1MB）
- 不要在每帧调用保存操作
- 大块数据建议在 `on_save` 钩子中进行保存
- 存档迁移应保持向后兼容
- 避免存储敏感信息（纯文本存储）
- 定期验证存档数据的完整性
