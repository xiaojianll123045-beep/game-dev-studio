# 脚本模组开发

脚本模组使用 GDScript 编写，通过 `bridge` 对象与游戏交互。本文档涵盖所有核心开发概念。

## 入口脚本

每个脚本模组需要在 `mod.json` 中指定一个入口脚本文件：

```json
{
  "id": "my_script_mod",
  "type": "script",
  "entry_point": "scripts/main.gd"
}
```

## 基础脚本结构

```gdscript
extends Node

# 模组初始化
func _ready():
    bridge.log_info("脚本模组已加载")
    bridge.register_hook("on_game_start", self, "_on_game_start")
    bridge.register_hook("on_tick", self, "_on_tick")
    bridge.register_command("my_command", self, "_on_my_command")

# 游戏启动时触发
func _on_game_start(data):
    bridge.log_info("游戏已启动，玩家: " + data.player_name)

# 每帧更新（慎用性能）
func _on_tick(delta):
    pass

# 自定义命令处理
func _on_my_command(args):
    bridge.log_info("执行命令，参数: " + str(args))
```

## bridge API 概览

所有游戏交互都通过 `bridge` 对象完成：

### 数据访问

```gdscript
# 获取游戏值
var value = bridge.get_game_value("economy.base_income")
var player_data = bridge.get_player_data(player_id)
var game_state = bridge.get_game_state()

# 设置游戏值
bridge.set_game_value("economy.base_income", 15)
```

### 日志系统

```gdscript
bridge.log_info("信息")
bridge.log_warn("警告")
bridge.log_error("错误")
bridge.log_debug("调试信息")  # 仅开发模式输出
```

### 钩子系统

```gdscript
# 注册钩子
bridge.register_hook("on_turn_end", self, "_on_turn_end")

# 取消注册
bridge.unregister_hook("on_turn_end", self)

# 触发自定义钩子
bridge.trigger_hook("my_custom_event", {"key": "value"})
```

### 命令系统

```gdscript
# 注册命令
bridge.register_command("give_gold", self, "_give_gold")

# 执行命令
bridge.execute_command("give_gold", {"amount": 1000})

func _give_gold(args):
    var player_id = bridge.get_current_player()
    bridge.add_resource(player_id, "gold", args.amount)
```

### 节点与场景

```gdscript
# 备用访问方式（当 bridge 全局不可用时）
var mod_bridge = get_node("/root/GameManager/ModBridge")
if mod_bridge:
    mod_bridge.log_info("通过节点路径访问桥接对象")

# 创建UI
var scene = bridge.load_scene("res://my_ui.tscn")
var instance = scene.instantiate()
bridge.add_child_to_layer(instance, "overlay")
```

## 键盘输入处理

在 Godot 4 中，使用 `_process` 配合 `Input.is_key_pressed()` 和去抖标志位来处理按键事件：

```gdscript
extends Node

var _last_pressed := {}  # 去抖标志字典

func _process(delta):
    # 检查 F5 按键（带去抖）
    if Input.is_key_pressed(KEY_F5) and not _last_pressed.get("f5", false):
        _last_pressed["f5"] = true
        _on_debug_toggle()
    elif not Input.is_key_pressed(KEY_F5):
        _last_pressed["f5"] = false

    # 检查组合键 Ctrl + Shift + D
    if Input.is_key_pressed(KEY_CTRL) and Input.is_key_pressed(KEY_SHIFT) and Input.is_key_pressed(KEY_D):
        bridge.log_debug("检测到调试组合键")

func _on_debug_toggle():
    bridge.log_info("切换调试模式")
    bridge.toggle_debug_overlay()

# 另一种写法：使用函数封装去抖逻辑
func is_key_just_pressed_custom(key: Key) -> bool:
    var key_name = "key_" + str(key)
    var pressed = Input.is_key_pressed(key)
    if pressed and not _last_pressed.get(key_name, false):
        _last_pressed[key_name] = true
        return true
    if not pressed:
        _last_pressed[key_name] = false
    return false

func _process(delta):
    if is_key_just_pressed_custom(KEY_F1):
        bridge.execute_command("show_help", {})

    if is_key_just_pressed_custom(KEY_ESCAPE):
        bridge.execute_command("close_ui", {})
```

## 模块组织

对于大型脚本模组，建议拆分文件：

```gdscript
# scripts/main.gd — 入口脚本
extends Node

func _ready():
    bridge.log_info("加载子模块...")
    var economy = preload("res://mods/my_mod/scripts/economy.gd").new()
    add_child(economy)

    var combat = preload("res://mods/my_mod/scripts/combat.gd").new()
    add_child(combat)

    bridge.register_hook("on_game_start", self, "_on_game_start")

func _on_game_start(data):
    bridge.log_info("所有子模块已就绪")
```

```gdscript
# scripts/economy.gd
extends Node

func _ready():
    bridge.register_hook("on_turn_end", self, "_on_turn_end")
    bridge.log_info("经济模块已加载")

func _on_turn_end(data):
    bridge.log_debug("回合结束 - 处理经济")
```

## 事件监听

```gdscript
func _ready():
    # 监听游戏事件
    bridge.register_hook("on_unit_destroyed", self, "_on_unit_destroyed")
    bridge.register_hook("on_city_founded", self, "_on_city_founded")
    bridge.register_hook("on_war_declared", self, "_on_war_declared")
    bridge.register_hook("on_trade_completed", self, "_on_trade_completed")
    bridge.register_hook("on_research_completed", self, "_on_research_completed")

func _on_unit_destroyed(data):
    bridge.log_info("单位被摧毁: " + data.unit_name)
    bridge.increment_stat("total_units_lost", 1)

func _on_city_founded(data):
    bridge.log_info("新城市建立: " + data.city_name)
    bridge.show_notification("新城市 " + data.city_name + " 已建立！")
```

## 性能优化

```gdscript
extends Node

var tick_counter := 0

func _ready():
    bridge.register_hook("on_tick", self, "_on_tick")

# 每帧调用，注意性能
func _on_tick(delta):
    tick_counter += 1
    # 每 60 帧执行一次耗时操作
    if tick_counter % 60 == 0:
        _heavy_operation()

func _heavy_operation():
    bridge.log_debug("执行定期检查...")
    # 耗时逻辑放在这里

# 使用定时器而不是每帧处理
func _ready():
    var timer = Timer.new()
    timer.wait_time = 5.0
    timer.timeout.connect(_periodic_check)
    add_child(timer)
    timer.start()

func _periodic_check():
    bridge.log_debug("每 5 秒执行一次检查")
```

## 错误处理

```gdscript
func _ready():
    # 安全调用
    var result = bridge.call("some_function")
    if result == null:
        bridge.log_warn("some_function 返回空值")

    # 使用 os.is_instance_valid 检查对象
    if bridge.is_instance_valid(some_ref):
        bridge.log_info("对象有效")

    # Try-catch 保护
    var success = bridge.safe_call("risky_operation", [arg1, arg2])
    if not success:
        bridge.log_error("危险操作失败")
```

## 注意事项

- 所有 API 调用使用 `bridge.xxx()` 而非 `ModAPI.xxx()`
- 键盘输入使用 `Input.is_key_pressed()` + 去抖标志，不要使用 `Input.is_key_just_pressed()`
- 入口脚本必须继承 `Node`
- 调用 `get_node("/root/GameManager/ModBridge")` 作为备用的桥接访问方式
- 注意控制 `_process` 中的逻辑复杂度
- 使用 `log_debug` 输出开发信息，正式版不会显示
- 模组数据使用 `save_mod_data`/`load_mod_data` 进行持久化
